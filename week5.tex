\section{Week 5}

Rij: $\{a_n\}_n\in\N$\\
Reeks: $\sum_{n=0}^\infty a_n:$ $a_n\in\R \vee a_n\in\C$\\
Meetkundige reeks: $\sum_{n=0}^\infty x^n$\\
Parti\"ele som: $S_n = \sum_{k=0}^n x^k$\\
\als{$S_n \xrightarrow{\infty} s$}
dan zeggen we de reeks $\sum_{n=0}^\infty x^n$
convergeert naar $s$
$\sum_{n=0}^\infty x^n = s$\\

\voorbeeld{
	$\sum_{n=0}^\infty x^n$\\
	$S_n = 1 + x + x^2 + x^3 ... = \frac{1-x^{n+1}}{1-x}$\\
	\bewijs{
		$(1-x)(1 + x + x^2 + x^3 ...) = (1 + x + x^2 + x^3 ...) - (x + x^2 + x^3 + x^4...) = 1-x^{n+1}$\\
	}
	\als{$\abs{x}<x$} dan $x^n+1 \to 0$\\
	\als{$S_n \xrightarrow{n \to \infty} \frac{1}{1-x}$}\\
}

\voorbeeld{
	$\sum_{k=1}^\infty\frac{1}{k(k+1)}$\\
	$S_n = \sum_{k=1}^n\frac{1}{k(k+1)} = \sum_{k=1}^n(\frac{1}{k}-\frac{1}{k+1})$\\
	$= (1+\frac{1}{2} + ... + \frac{1}{n}) - (\frac{1}{2} + \frac{1}{3} + ... + \frac{1}{n + 1})=1-\frac{1}{n+1}$\\
	$\sum_{k=1}^\infty\frac{1}{k(k+1)} = 1$
}
