\section{Week 1}
\definitie{totale ordening}{
	\als{$\leq$ partiele ordening}{
		$\forall x: x \leq x$ (reflexief)\\
		$\forall x,y: x \leq y \wedge y \leq z \to x \leq z$ (transitief)\\
		$\forall x,y: x \leq y \wedge y \leq x \to x = y$ (antisymetrish)\\
	}
	\als{$\leq$ is totaal}{
		$\forall x,y: x \leq y \wedge y \leq x$
	}
}
